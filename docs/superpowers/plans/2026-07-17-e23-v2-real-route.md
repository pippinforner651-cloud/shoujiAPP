# E23 V2 Real China Loop Route Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build and audit a versioned static 27,000+ kilometre real-road China loop from Shenzhen PKU HSBC Business School back to the same campus, while keeping it non-active until publication is explicitly approved.

**Architecture:** Use a provider-neutral route-generation pipeline that turns an auditable waypoint corridor into a static route package. Candidate geometry may be generated by an offline build-time routing source, but runtime never calls routing or tile services. Automated gates validate topology, distance, provenance, content coverage, hashes, and rollback metadata; manual segment review is required before `VERIFIED`.

**Tech Stack:** Node.js 22, TypeScript, JSON Schema 2020-12, GeoJSON LineString, ECharts China GeoJSON for offline audit display, SHA256 manifests, Node `node:test`.

## Corridor Decision

Official public references describe:

- G228 Dandong–Dongxing coastal corridor: approximately 7,378 km.
- G331 Dandong–Altay northern border corridor: approximately 9,333 km.
- G219 Kanas–Dongxing western/southern border corridor: approximately 10,065 km.

The three published figures total about 26,776 km, below the product minimum. The route therefore includes a real Qiongzhou Strait vehicle ferry and a real Hainan G98 island loop before returning to the G228 corridor. The measured static geometry, not the published approximate figures, determines the final package distance.

## Failure Rollback

- Route versions are immutable after `PUBLISHED`; corrections create a new version.
- Activity binding stores exact `routePackageId + version + distancePolicy`.
- A bad candidate remains `DRAFT` and cannot be loaded by production route repository.
- Reverting a route commit never deletes activities, workouts, or contributions.
- Raw routing responses and audit files are retained under `D:\CODEX制作文件\E23跑起来_V2真实路线制作_20260717`.
- If a source disappears, retain its captured hash and mark affected segments `REVIEW_REQUIRED`; never silently replace the evidence.

- [ ] **Step 3: Implement the parse-only adapter**

```js
export function parseOsrmRoute(payload) {
  if (payload?.code !== 'Ok' || !payload.routes?.[0]) {
    throw new Error(`routing failed: ${payload?.code ?? 'UNKNOWN'}`);
  }
  const route = payload.routes[0];
  const coordinates = route.geometry?.coordinates ?? [];
  const roadNames = [...new Set(
    (route.legs ?? [])
      .flatMap((leg) => leg.steps ?? [])
      .map((step) => String(step.name ?? '').trim())
      .filter(Boolean),
  )];
  if (coordinates.length < 2) throw new Error('routing geometry is empty');
  return {
    distanceMeters: Math.round(route.distance),
    coordinates,
    roadNames,
    rawEvidence: payload,
  };
}
```

```ts
export interface RouteArtifactValidationInput {
  route: RoutePackageV2;
  sourceKindsById: Record<string, 'OFFICIAL_PLACE' | 'OFFICIAL_CORRIDOR' | 'OFFICIAL_FERRY' | 'ROUTING_EVIDENCE'>;
  reviewStatusBySegment: Record<string, 'PASS' | 'REVIEW_REQUIRED' | 'REJECTED'>;
  previousPublishedRouteSha256: string | null;
}

export function validateRouteArtifacts(
  input: RouteArtifactValidationInput,
): RouteValidationIssue[] {
  const issues: RouteValidationIssue[] = [];
  const add = (code: string, path: string, message: string) =>
    issues.push({ code, path, message });
  const isHash = (value: string) => /^[a-f0-9]{64}$/i.test(value);
  const isFinal = ['VERIFIED', 'PUBLISHED', 'ACTIVE'].includes(input.route.status);

  if (isFinal && !isHash(input.route.integrity.evidenceManifestSha256)) {
    add('MISSING_EVIDENCE_HASH', 'integrity.evidenceManifestSha256', '缺少有效证据哈希');
  }
  if (isFinal && !isHash(input.route.integrity.contentPackageSha256)) {
    add('MISSING_CONTENT_HASH', 'integrity.contentPackageSha256', '缺少有效内容哈希');
  }

  input.route.segments.forEach((segment, index) => {
    for (const sourceRef of segment.sourceRefs) {
      if (!input.sourceKindsById[sourceRef]) {
        add('UNKNOWN_SOURCE_REF', `segments.${index}.sourceRefs`, `未知来源 ${sourceRef}`);
      }
    }
    if (segment.kind === 'ROAD' && !segment.roadNames.some((name) => name.trim().length >= 2)) {
      add('ROAD_NAME_REQUIRED', `segments.${index}.roadNames`, '道路段必须有真实道路名称');
    }
    if (segment.kind === 'FERRY' && !segment.sourceRefs.some(
      (sourceRef) => input.sourceKindsById[sourceRef] === 'OFFICIAL_FERRY',
    )) {
      add('OFFICIAL_FERRY_SOURCE_REQUIRED', `segments.${index}.sourceRefs`, '轮渡段必须有官方来源');
    }
    if (segment.distanceMeters > 300_000) {
      add('SEGMENT_REQUIRES_INTERMEDIATE_POINT', `segments.${index}`, '超过300公里需增加审计节点');
    }
    if (isFinal && input.reviewStatusBySegment[segment.id] !== 'PASS') {
      add('SEGMENT_REVIEW_INCOMPLETE', `segments.${index}`, '路段人工审核未通过');
    }
  });

  if (
    input.route.status === 'PUBLISHED'
    && input.previousPublishedRouteSha256
    && input.previousPublishedRouteSha256 !== input.route.integrity.routeSha256
  ) {
    add('PUBLISHED_VERSION_MUTATED', 'integrity.routeSha256', '已发布版本不可原地修改');
  }
  return issues;
}
```

```ts
interface StoryIssue { code: string; path: string; message: string }
interface StoryItem { name: string; summary: string; sourceRefs: string[] }
interface StoryInterval {
  storyPointId: string;
  segmentId: string;
  fromMeters: number;
  toMeters: number;
  roadNames: string[];
  attractions: StoryItem[];
  foods: StoryItem[];
  disclaimer: string;
}
interface StorySegment { id: string; distanceMeters: number; roadNames: string[] }

export function validateRouteStories(
  segments: StorySegment[],
  stories: StoryInterval[],
  sourceIds: Set<string>,
): StoryIssue[] {
  const issues: StoryIssue[] = [];
  const add = (code: string, path: string, message: string) =>
    issues.push({ code, path, message });
  const forbidden = /实时导航|实时排名|当前热门|排名第一/;

  segments.forEach((segment, segmentIndex) => {
    const intervals = stories
      .filter((story) => story.segmentId === segment.id)
      .sort((left, right) => left.fromMeters - right.fromMeters);
    let cursor = 0;
    intervals.forEach((story, storyIndex) => {
      const path = `segments.${segmentIndex}.stories.${storyIndex}`;
      if (story.fromMeters !== cursor || story.toMeters <= story.fromMeters) {
        add('STORY_COVERAGE_GAP', path, '地点内容区间未连续覆盖路线');
      }
      cursor = story.toMeters;
      if (story.roadNames.some((name) => !segment.roadNames.includes(name))) {
        add('STORY_ROAD_MISMATCH', `${path}.roadNames`, '地点内容道路与路线段不一致');
      }
      for (const item of [...story.attractions, ...story.foods]) {
        if (item.sourceRefs.some((sourceRef) => !sourceIds.has(sourceRef))) {
          add('STORY_SOURCE_UNKNOWN', path, '地点内容包含未知来源');
        }
        if (forbidden.test(`${item.name}${item.summary}`)) {
          add('STORY_LIVE_CLAIM_FORBIDDEN', path, '静态内容不得声称实时导航或实时排名');
        }
      }
    });
    if (cursor !== segment.distanceMeters) {
      add('STORY_COVERAGE_GAP', `segments.${segmentIndex}`, '地点内容未覆盖完整路段');
    }
  });
  return issues;
}
```
